160 Intersection of Two Linked Lists (Kieran)

Write a program to find the node at which the intersection of two singly linked lists begins.

class Solution:
    def getIntersectionNode(self, headA: ListNode, headB: ListNode) -> ListNode:
        visited = set()
        while headA or headB:
            if headA:
                if headA in visited:
                    return headA
                else:
                    visited.add(headA)
                headA=headA.next
            if headB:
                if headB in visited:
                    return headB
                else:
                    visited.add(headB)
                headB=headB.next

        return None
